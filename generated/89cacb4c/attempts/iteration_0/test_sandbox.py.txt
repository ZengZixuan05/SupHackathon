from fastapi.testclient import TestClient
import pytest
from sandbox_app import app

client = TestClient(app)

@pytest.fixture(autouse=True)
def clear_store():
    import sandbox_app
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()
    yield
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()

def test_user_registration_success():
    response = client.post("/api/users/register", json={
        "username": "testuser1",
        "email": "testuser1@example.com",
        "password": "password123"
    })
    assert response.status_code == 201
    data = response.json()
    assert "id" in data
    assert data["username"] == "testuser1"
    assert data["email"] == "testuser1@example.com"

def test_like_post_success():
    client.post("/api/users/register", json={
        "username": "testuser12",
        "email": "testuser12@example.com",
        "password": "password123"
    })
    post_response = client.post("/api/posts", json={
        "user_id": 1,
        "content": "Post to like."
    })
    post_id = post_response.json()["post_id"]
    response = client.post(f"/api/posts/{post_id}/like", json={

    assert data["post_id"] == post_id
