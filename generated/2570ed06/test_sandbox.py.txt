client = TestClient(app)

        sandbox_app.store.clear()

def test_user_registration_success():
    response = client.post("/api/users/register", json={
        "username": "testuser1",
        "email": "testuser1@example.com",
        "password": "password123"
    })
    assert response.status_code == 201
    assert "message" in response.json()

def test_add_comment_success():
    client.post("/api/users/register", json={
        "username": "testuser10",
        "email": "testuser10@example.com",
        "password": "password123"
    })
    user_response = client.post("/api/users/login", json={
        "email": "testuser10@example.com",
        "password": "password123"
    })
    user_id = user_response.json()["user_id"]
    post_response = client.post("/api/posts", json={
        "user_id": user_id,
        "content": "This is a test post for comments."
    })
    post_id = post_response.json()["post_id"]
    response = client.post(f"/api/posts/{post_id}/comments", json={
        "user_id": user_id,
        "comment": "This is a test comment."
    })
    assert response.status_code == 201
    assert "message" in response.json()
    assert "comment_id" in response.json()

def test_send_friend_request_success():
    client.post("/api/users/register", json={
        "username": "testuser11",
        "email": "testuser11@example.com",
        "password": "password123"
    })
    client.post("/api/users/register", json={
        "username": "testuser12",
        "email": "testuser12@example.com",
        "password": "password123"
    })
    user_response_1 = client.post("/api/users/login", json={
        "email": "testuser11@example.com",
        "password": "password123"
    })
    user_id_1 = user_response_1.json()["user_id"]
    user_response_2 = client.post("/api/users/login", json={
        "email": "testuser12@example.com",
        "password": "password123"
    })
    user_id_2 = user_response_2.json()["user_id"]
    response = client.post("/api/friends/request", json={
        "user_id": user_id_1,
        "friend_id": user_id_2
    })
    assert response.status_code == 200
    assert "message" in response.json()
