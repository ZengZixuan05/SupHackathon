from fastapi import FastAPI, HTTPException
from pydantic import BaseModel, constr, conint, validator
from typing import List, Dict

app = FastAPI()
store: Dict[str, dict] = {}

class ItemCreate(BaseModel):
    name: str
    sku: constr(min_length=1)  # SKU must be a non-empty string
    quantity: conint(ge=0)      # Quantity must be greater than or equal to 0

    @validator('sku')
    def sku_must_be_unique(cls, sku):
        if sku in store:
            raise ValueError("SKU already exists")
        return sku

class ItemResponse(BaseModel):
    name: str
    sku: str
    quantity: int

class SellItemRequest(BaseModel):
    amount: conint(ge=1)         # Amount must be greater than or equal to 1

class ItemResult(BaseModel):
    message: str
    item: ItemResponse

@app.post("/items", response_model=ItemResult, status_code=201)
def add_item(item: ItemCreate):
    store[item.sku] = item.dict()
    return ItemResult(message="Item added successfully", item=ItemResponse(**store[item.sku]))

@app.get("/items", response_model=List[ItemResponse])
def list_items():
    return [ItemResponse(**item) for item in store.values()]

@app.get("/items/{sku}", response_model=ItemResponse)
def get_item(sku: str):
    if sku not in store:
        raise HTTPException(status_code=404, detail="Item not found")
    return ItemResponse(**store[sku])

@app.post("/items/{sku}/sell", response_model=ItemResult)
def sell_item(sku: str, sell_request: SellItemRequest):
    if sku not in store:
        raise HTTPException(status_code=404, detail="Item not found")
    
    item = store[sku]
    if item['quantity'] < sell_request.amount:
        raise HTTPException(status_code=400, detail="Insufficient stock")
    
    item['quantity'] -= sell_request.amount
    return ItemResult(message="Stock decremented successfully", item=ItemResponse(**item))
