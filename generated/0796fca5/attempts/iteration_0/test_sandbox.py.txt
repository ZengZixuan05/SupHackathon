from fastapi.testclient import TestClient
import pytest
from sandbox_app import app

client = TestClient(app)

def clear_store():
    import sandbox_app
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()
    yield
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()

def test_user_registration_success():
    response = client.post("/api/v1/users/register", json={
        "username": "testuser1",
        "email": "testuser1@example.com",
        "password": "password123"
    })
    assert response.status_code == 201
    assert "user_id" in response.json()
    assert response.json()["message"] == "User registered successfully."

def test_create_post_success():
    client.post("/api/v1/users/register", json={
        "username": "testuser7",
        "email": "testuser7@example.com",
        "password": "password123"
    })
    login_response = client.post("/api/v1/users/login", json={
        "email": "testuser7@example.com",
        "password": "password123"
    })
    user_id = login_response.json()["user_id"]
    response = client.post("/api/v1/posts", json={
        "user_id": user_id,
        "content": "This is a test post."
    })
    assert response.status_code == 201
    assert "post_id" in response.json()
    assert response.json()["message"] == "Post created successfully."

    })

def test_create_post_long_content():
    client.post("/api/v1/users/register", json={
        "username": "testuser8",
        "email": "testuser8@example.com",
        "password": "password123"
    })
    login_response = client.post("/api/v1/users/login", json={
        "email": "testuser8@example.com",
        "password": "password123"
    })
    user_id = login_response.json()["user_id"]
    long_content = "a" * 1001  # 1001 characters
    response = client.post("/api/v1/posts", json={
        "user_id": user_id,
        "content": long_content
    })
    assert response.status_code == 400
    assert "detail" in response.json()

def test_get_user_posts_success():
    client.post("/api/v1/users/register", json={
        "username": "testuser9",
        "email": "testuser9@example.com",
        "password": "password123"
    })
    login_response = client.post("/api/v1/users/login", json={
        "email": "testuser9@example.com",
        "password": "password123"
    })
    user_id = login_response.json()["user_id"]
    client.post("/api/v1/posts", json={
        "user_id": user_id,
        "content": "First post."
    })
    response = client.get(f"/api/v1/users/{user_id}/posts")
    assert response.status_code == 200
    assert "posts" in response.json()
    assert isinstance(response.json()["posts"], list)

def test_get_user_posts_non_existent_user():
    response = client.get("/api/v1/users/nonexistent_user/posts")
    assert response.status_code == 404
    assert "detail" in response.json()
