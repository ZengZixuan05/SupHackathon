import pytest
from fastapi.testclient import TestClient
from sandbox_app import app

client = TestClient(app)

def clear_store():
    import sandbox_app
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()
    yield
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()

def test_user_registration_success():
    response = client.post("/api/v1/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    assert response.status_code == 201
    assert "user_id" in response.json()
    assert response.json()["message"] == "User registered successfully."

def test_user_registration_conflict_username():
    client.post("/api/v1/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    response = client.post("/api/v1/users/register", json={
        "username": "testuser",
        "email": "newuser@example.com",
        "password": "securepassword"
    })
    assert response.status_code == 409
    assert "detail" in response.json()

def test_user_login_success():
    client.post("/api/v1/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    response = client.post("/api/v1/users/login", json={
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    assert response.status_code == 200
    assert "token" in response.json()
    assert "user_id" in response.json()

def test_user_login_invalid_credentials():
    response = client.post("/api/v1/users/login", json={
        "email": "nonexistent@example.com",
        "password": "wrongpassword"
    })
    assert response.status_code == 401
    assert "detail" in response.json()

def test_create_post_success():
    client.post("/api/v1/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    user_response = client.post("/api/v1/users/login", json={
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    user_id = user_response.json()["user_id"]
    
    response = client.post("/api/v1/posts", json={
        "user_id": user_id,
        "content": "This is a test post."
    })
    assert response.status_code == 201
    assert "post_id" in response.json()
    assert response.json()["message"] == "Post created successfully."

def test_create_post_user_not_found():
    response = client.post("/api/v1/posts", json={
        "user_id": "nonexistent_user",
        "content": "This post should fail."
    })
    assert response.status_code == 404
    assert "detail" in response.json()

def test_get_user_posts_success():
    client.post("/api/v1/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    user_response = client.post("/api/v1/users/login", json={
        "email": "testuser@example.com",
        "password": "securepassword"
    })
    user_id = user_response.json()["user_id"]
    
    client.post("/api/v1/posts", json={
        "user_id": user_id,
        "content": "This is a test post."
    })
    
    response = client.get(f"/api/v1/users/{user_id}/posts")
    assert response.status_code == 200
    assert isinstance(response.json(), list)
    assert len(response.json()) > 0

def test_get_user_posts_not_found():
    response = client.get("/api/v1/users/nonexistent_user/posts")
    assert response.status_code == 404
    assert "detail" in response.json()
