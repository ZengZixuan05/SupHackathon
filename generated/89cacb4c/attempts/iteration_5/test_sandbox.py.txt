import pytest
from fastapi.testclient import TestClient
from sandbox_app import app

client = TestClient(app)

@pytest.fixture(autouse=True)
def clear_store():
    import sandbox_app
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()
    if hasattr(sandbox_app, "posts"):
        sandbox_app.posts.clear()
    yield
    if hasattr(sandbox_app, "store"):
        sandbox_app.store.clear()
    if hasattr(sandbox_app, "posts"):
        sandbox_app.posts.clear()

def test_register_user_success():
    response = client.post("/api/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "password123"
    })
    assert response.status_code == 201
    assert "id" in response.json()
    assert response.json()["username"] == "testuser"
    assert response.json()["email"] == "testuser@example.com"

def test_register_user_existing_username():
    client.post("/api/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "password123"
    })
    response = client.post("/api/users/register", json={
        "username": "testuser",
        "email": "newuser@example.com",
        "password": "password123"
    })
    assert response.status_code == 409
    assert "detail" in response.json()

def test_login_user_success():
    client.post("/api/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "password123"
    })
    response = client.post("/api/users/login", json={
        "email": "testuser@example.com",
        "password": "password123"
    })
    assert response.status_code == 200
    assert "token" in response.json()
    assert "user_id" in response.json()

    })
    client.post("/api/posts", json={

def test_like_post_success():
    client.post("/api/users/register", json={
        "username": "testuser",
        "email": "testuser@example.com",
        "password": "password123"
    })
    post_response = client.post("/api/posts", json={
        "user_id": 1,
        "content": "This is a test post."
    })
    response = client.post(f"/api/posts/{post_response.json()['post_id']}/like", json={
        "user_id": 1
    })
    assert response.status_code == 200
    assert response.json()["post_id"] == post_response.json()["post_id"]
    assert response.json()["likes_count"] == 1

def test_comment_on_post_success():
    client.post("/api/users/register", json={
